・Union-Findテンプレ(コピペ)


struct UnionFind {
  vector<int> par, sz;
  UnionFind(int n) : par(n), sz(n,1) {
    for (int i = 0; i < n; i++) par[i] = i;
  }
  int root(int x) {
    return par[x] == x ? x : par[x] = root(par[x]);
  }
  bool same(int a, int b) {
    return root(a) == root(b);
  }
  void unite(int a, int b) {
    a = root(a); b = root(b);
    if (a == b) return;
    if (sz[a] < sz[b]) swap(a,b);
    par[b] = a;
    sz[a] += sz[b];
  }
  int size(int x) {
    return sz[root(x)];
  }
};

・重み付きUnion-Find(コピペ)


struct WUnionFind {
  vector<int> par, sz, weight;
  UnionFind(int n) : par(n), sz(n,1), weight(n,0) {
    for (int i = 0; i < n; i++) par[i] = i;
  }
  int root(int x) {
    if (par[x] == x) return x;
    int r = root(par[x]);
    weight[x] += weight[par[x]];
    return par[x] = r;
  }
  int weight_to_root(int x) {
    root(x);
    return weight[x];
  }
  bool same(int a, int b) {
    return root(a) == root(b);
  }
  bool unite(int a, int b, int w) {
    w += weight_to_root(a); w -= weight_to_root(b);
    a = root(a); b = root(b);
    if (a == b) return false;
    if (sz[a] < sz[b]) swap(a,b), w = -w;
    par[b] = a;
    weight[b] = w;
    sz[a] += sz[b];
    return true;
  }
  int size(int x) {
    return sz[root(x)];
  }
  int diff(int a, int b) {
    return weight_to_root(b) - weight_to_root(a);
  }
};

・Kruskal(最小全域木を求める)*UnionFind前提*
問題文で連結が保証されている場合も多い 


int main() {
  int N,M; cin >> N >> M;
  UnionFind uf(N);
  vector> edges(M);
  for (int i = 0; i < M; i++) {
    int a,b; ll w; cin >> a >> b >> w;
    a--; b--;
    edges[i] = {w,a,b};
  }
  sort(edges.begin(),edges.end());
  int used = 0;
  ll ans = 0;
  for (auto [w,a,b] : edges) {
    if (used == N-1) break;
    if (uf.same(a,b)) continue;
    uf.unite(a,b); ans += w; used++;
  }
  if (used != N-1) cout << -1 << endl;
  else cout << ans << endl;
}

・DFS(全頂点を巡るだけ)


Graph G;
vector<int> visited;

void dfs(int v) {
  visited[v] = 1;
  for (int nv : G[v]) {
      if (!visited[nv]) dfs(nv);
  }
}

int main() {
  int N,M; cin >> N >> M;
  G.resize(N); visited.assign(N,0);
  for (int i = 0; i < M; i++) {
    int a,b; cin >> a >> b;
    a--; b--;
    G[a].push_back(b);
    G[b].push_back(a); //有効グラフならいらない
  }
  dfs(0); //0スタート設定、入力によって変更
}

・BFS(各頂点への最短距離)


int main() {
  int N,M; cin >> N >> M;
  Graph G(N); vector<int> dist(N,-1);
  for (int i = 0; i < M; i++) {
    int a,b; cin >> a >> b;
    a--; b--;
    G[a].push_back(b);
    G[b].push_back(a); //有効グラフならいらない
  }
  queue<int> que;
  que.push(0); dist[0] = 0; //0スタート設定、入力によって変更
  while (!que.empty()) {
    int v = que.front(); que.pop();
    for (int nv : G[v]) {
      if (dist[nv] != -1) continue;
      dist[nv] = dist[v]+1;
      que.push(nv);
    }
  }
}

・dijkstra(重み付きの最短距離)


int main() {
  int N,M; cin >> N >> M;
  WGraph G(N); vector<ll> dist(N,INF);
  for (int i = 0; i < M; i++) {
    int a,b,c; cin >> a >> b >> c;
    a--; b--;
    G[a].push_back({b,c});
    G[b].push_back({a,c}); //有効グラフならいらない
  }
  priority_queue<pair<ll,int>,vector<pair<ll,int>>,greater<pair<ll,int>>> pq;
  pq.push({0,0}); dist[0]  = 0; //0スタート設定、入力によって変更
  while (!pq.empty()) {
    auto [d,v] = pq.top(); pq.pop();
    if (d > dist[v] || dist[v] == INF) continue;
    for (auto [nv,cost] : G[v]) {
      if (dist[nv] > dist[v] + cost) {
        dist[nv] = dist[v] + cost;
        pq.push({dist[nv],nv});
      }
    }
  }
}
